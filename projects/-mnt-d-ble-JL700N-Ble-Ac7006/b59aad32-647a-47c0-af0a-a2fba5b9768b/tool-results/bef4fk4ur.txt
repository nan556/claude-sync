+void GetCrC(u8 *pSource, u32 len, u8 *crc)
+{
+u16 crc16 = 0xFFFF;
+u32 i = 0;
+u32 j = 0;
+if(len > 0xFFFFFFFF-1) return;
+while(j < len)
+{
+	crc16 ^= pSource[j];
+	for(i = 0; i < 8; i++)
+	{
+	if(crc16 & 0x01)
+	{
+	crc16 >>= 1;
+	crc16 ^= 0xA001;
+	}
+	else
+	{
+	crc16 >>= 1;
+	}
+	}
+	j++;
+}
+	crc[0] = crc16 % 0x100;
+	crc[1] = crc16 / 0x100;
+}
